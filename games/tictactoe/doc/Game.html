def finish
        if @@turn == 9
                puts "It was a tie D:"
        end
        puts 'Do you want to start again?(yes/no)'
        input = gets.chomp
        if (input == 'no')
                puts 'Bye bye'
        elsif (input == 'yes')
                puts 'Let\s start again'
                @board = Board.new
                @@turn = 0
                play
        end
end

    # File tictactoe.rb, line 23
def play
        @board.print_grid
        do_break = false
        while !((@board.checkWin('X', @p1.name)) || (@board.checkWin('O', @p2.name)))
                player_turn
                if @@turn == 9
                        do_break = true
                end
                break if do_break
        end
        finish
end

    # File tictactoe.rb, line 37
def player_turn
        @@turn += 1
        puts "It is the turn of player " + (@@player+1).to_s
        puts "What row do you want to update?"
        row = gets.chomp.to_i
        puts "What column do you want to update?"
        column = gets.chomp.to_i
        if @@player == 0
                @board.update(row, column, @p1.sym)
                @@player = 1
        else
                @board.update(row, column, @p2.sym)
                @@player = 0
        end

        @board.print_grid
end
